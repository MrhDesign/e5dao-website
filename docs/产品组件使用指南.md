### 4. 搜索结果页面

#### 4.1 产品搜索功能
```typescript
const ProductSearchPage = () => {
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedCategory, setSelectedCategory] = useState<number | null>(null);
  const [filteredProducts, setFilteredProducts] = useState<Product[]>([]);
  
  const { getContent } = useContent();
  const productsData = getContent<Product[]>('products.items') || [];
  const categoriesData = getContent<ProductCategory[]>('products.categories') || [];

  // 搜索过滤逻辑
  useEffect(() => {
    let results = productsData;
    
    // 分类筛选
    if (selectedCategory) {
      results = getProductsByCategory(results, selectedCategory);
    }
    
    // 关键词搜索
    if (searchQuery.trim()) {
      results = results.filter(product => {
        const searchText = searchQuery.toLowerCase();
        const productTitle = 'title' in product ? product.title : '';
        const productModel = 'model' in product ? product.model : '';
        const productDescription = 'description' in product ? product.description : '';
        
        return (
          productTitle.toLowerCase().includes(searchText) ||
          productModel.toLowerCase().includes(searchText) ||
          productDescription?.toLowerCase().includes(searchText)
        );
      });
    }
    
    setFilteredProducts(results);
  }, [searchQuery, selectedCategory, productsData]);

  return (
    <div className="container mx-auto px-5 py-10">
      {/* 搜索和筛选区域 */}
      <div className="mb-8">
        <div className="flex flex-col lg:flex-row gap-4">
          {/* 搜索框 */}
          <input
            type="text"
            placeholder="搜索产品..."
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className="flex-1 px-4 py-2 border border-gray-300 rounded-lg"
          />
          
          {/* 分类筛选 */}
          <select
            value={selectedCategory || ''}
            onChange={(e) => setSelectedCategory(e.target.value ? Number(e.target.value) : null)}
            className="px-4 py-2 border border-gray-300 rounded-lg"
          >
            <option value="">所有分类</option>
            {categoriesData.map(category => (
              <option key={category.id} value={category.id}>
                {category.title}
              </option>
            ))}
          </select>
        </div>
      </div>
      
      {/* 搜索结果 */}
      <div className="mb-4">
        <p className="text-gray-600">
          找到 {filteredProducts.length} 个产品
          {searchQuery && ` 包含 "${searchQuery}"`}
        </p>
      </div>
      
      {/* 产品结果展示 */}
      <div className="grid grid-cols-2 lg:grid-cols-4 gap-2.5 lg:gap-10">
        {filteredProducts.map((product) => (
          <ProductCard
            key={product.id}
            product={product}
            className="w-full"
          />
        ))}
      </div>
      
      {/* 空搜索结果 */}
      {filteredProducts.length === 0 && (
        <div className="text-center py-20">
          <p className="text-gray-500">未找到匹配的产品</p>
          <p className="text-sm text-gray-400 mt-2">请尝试其他关键词或筛选条件</p>
        </div>
      )}
    </div>
  );
};
```

const processProductImages = (product: Product): string[] => {
  const images: string[] = [];
  
  // 添加主图
  if (product.image && product.image.trim() !== '') {
    images.push(product.image);
  }
  
  // 添加画廊图片
  if (product.gallery && Array.isArray(product.gallery)) {
    const validGalleryImages = product.gallery.filter(img => 
      img && typeof img === 'string' && img.trim() !== ''
    );
    images.push(...validGalleryImages);
  }
  
  return images;
};

// 使用示例
const ProductDetail = ({ product }: { product: Product }) => {

    );
  }
  
  return (

### 1. 路由验证

#### 1.1 分类URL验证
```typescript
// 验证产品详情页URL的分类是否正确
const validateProductUrl = (
  productId: number,
  categorySlug: string,
  products: Product[],
  categories: ProductCategory[]
): boolean => {
  // 1. 找到产品
  const product = products.find(p => p.id === productId);
  if (!product) return false;
  
  // 2. 获取产品的实际分类
  const productCategory = getCategoryByProduct(product, categories);
  if (!productCategory) return false;
  
  // 3. 获取URL中的分类
  const urlCategory = getCategoryBySlug(categorySlug, categories);
  if (!urlCategory) return false;
  
  // 4. 验证一致性
  return productCategory.id === urlCategory.id;
};

// 在页面组件中使用
const ProductDetailPage = () => {
  const params = useParams();
  const productId = parseInt(params.id as string);
  const categorySlug = params.category as string;
  
  const { getContent } = useContent();
  const productsData = getContent<Product[]>('products.items') || [];
  const categoriesData = getContent<ProductCategory[]>('products.categories') || [];
  
  // 验证URL有效性
  if (!validateProductUrl(productId, categorySlug, productsData, categoriesData)) {
    return <div>页面不存在</div>;
  }
  
  // 正常渲染页面
  return <div>产品详情页</div>;
};
```

### 3. 产品推荐系统

#### 3.1 智能相关产品推荐
```typescript
const RelatedProductsSection = ({ currentProduct }: { currentProduct: Product }) => {
  const { getContent } = useContent();
  const productsData = getContent<Product[]>('products.items') || [];
  const categoriesData = getContent<ProductCategory[]>('products.categories') || [];
  
  // 获取相关产品的智能推荐算法
  const getRelatedProducts = useMemo(() => {
    // 1. 获取同分类产品
    const categoryProducts = getProductsByCategory(productsData, currentProduct.categoryId);
    
    // 2. 获取当前产品在分类中的位置
    const currentIndex = categoryProducts.findIndex(p => p.id === currentProduct.id);
    
    // 3. 构建推荐产品数组
    const relatedProducts: Product[] = [];
    const maxRecommendations = 6;
    
    // 添加前一个产品（如果存在）
    if (currentIndex > 0) {
      relatedProducts.push(categoryProducts[currentIndex - 1]);
    }
    
    // 添加后续产品
    for (let i = currentIndex + 1; i < categoryProducts.length && relatedProducts.length < maxRecommendations; i++) {
      relatedProducts.push(categoryProducts[i]);
    }
    
    // 如果还有空间，从前面补充更多产品
    if (relatedProducts.length < maxRecommendations) {
      for (let i = currentIndex - 2; i >= 0 && relatedProducts.length < maxRecommendations; i--) {
        relatedProducts.unshift(categoryProducts[i]);
      }
    }
    
    return relatedProducts;
  }, [currentProduct, productsData]);
  
  if (getRelatedProducts.length === 0) {
    return null;
  }
  
  return (
    <div className="mt-8">
      <h2 className="text-2xl font-bold mb-6">相关产品</h2>
      <div className="grid grid-cols-2 lg:grid-cols-3 gap-4">
        {getRelatedProducts.map((product) => (
          <ProductCard
            key={product.id}
            product={product}
            className="w-full"
          />
        ))}
      </div>
    </div>
  );
};
```

// 使用虚拟列表优化大量产品展示
import { FixedSizeGrid } from 'react-window';

const VirtualizedProductGrid = ({ products }: { products: Product[] }) => {
  const COLUMN_COUNT = 4;
  const ITEM_WIDTH = 250;
  const ITEM_HEIGHT = 350;

  const GridItem = ({ 
    columnIndex, 
    rowIndex, 
    style 
  }: {
    columnIndex: number;
    rowIndex: number;
    style: React.CSSProperties;
  }) => {
    const productIndex = rowIndex * COLUMN_COUNT + columnIndex;
    const product = products[productIndex];
    
    if (!product) return <div style={style} />;
    
    return (
      <div style={style} className="p-2">
        <ProductCard product={product} />
      </div>
    );
  };

  return (
    <FixedSizeGrid
      columnCount={COLUMN_COUNT}
      columnWidth={ITEM_WIDTH}
      height={600}
      rowCount={Math.ceil(products.length / COLUMN_COUNT)}
      rowHeight={ITEM_HEIGHT}
      width="100%"
    >
      {GridItem}
    </FixedSizeGrid>
  );
};
